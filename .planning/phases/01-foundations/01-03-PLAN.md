Preload bridge (launcher/src/preload/index.ts — RESEARCH.md §Pattern 1 Two-process IPC):
```typescript
import { contextBridge, ipcRenderer } from 'electron';

type Unsubscribe = () => void;

contextBridge.exposeInMainWorld('wiiwho', {
  auth: {
    status: () => ipcRenderer.invoke('auth:status'),
    login: () => ipcRenderer.invoke('auth:login'),
    logout: () => ipcRenderer.invoke('auth:logout'),
    onDeviceCode: (cb: (p: unknown) => void): Unsubscribe => {
      const h = (_: unknown, p: unknown) => cb(p);
      ipcRenderer.on('auth:device-code', h);
      return () => ipcRenderer.off('auth:device-code', h);
    },
  },
  game: {
    play: () => ipcRenderer.invoke('game:play'),
    cancel: () => ipcRenderer.invoke('game:cancel'),
    status: () => ipcRenderer.invoke('game:status'),
    onStatus: (cb: (s: unknown) => void): Unsubscribe => {
      const h = (_: unknown, s: unknown) => cb(s);
      ipcRenderer.on('game:status-changed', h);
      return () => ipcRenderer.off('game:status-changed', h);
    },
    onProgress: (cb: (p: unknown) => void): Unsubscribe => {
      const h = (_: unknown, p: unknown) => cb(p);
      ipcRenderer.on('game:progress', h);
      return () => ipcRenderer.off('game:progress', h);
    },
  },
  settings: {
    get: () => ipcRenderer.invoke('settings:get'),
    set: (patch: unknown) => ipcRenderer.invoke('settings:set', patch),
  },
  logs: {
    readCrash: (opts?: { crashId?: string }) => ipcRenderer.invoke('logs:read-crash', opts ?? {}),
  },
  __debug: {
    securityAudit: () => ipcRenderer.invoke('__security:audit'),
  },
});
```

`__security:audit` handler (launcher/src/main/ipc/security.ts — RESEARCH.md §Electron Security Runtime Verification):
```typescript
import { ipcMain, BrowserWindow } from 'electron';

export function registerSecurityHandlers() {
  ipcMain.handle('__security:audit', () => {
    const win = BrowserWindow.getAllWindows()[0];
    const prefs = win?.webContents.getWebPreferences();
    const contextIsolation = prefs?.contextIsolation === true;
    const nodeIntegration = prefs?.nodeIntegration === false; // inverted: we want this to BE false
    const sandbox = prefs?.sandbox === true;
    return {
      contextIsolation,
      nodeIntegration, // reported as: "is it safe?" so true means nodeIntegration is OFF as desired
      sandbox,
      allTrue: contextIsolation && nodeIntegration && sandbox,
    };
  });
}
```

  auth: {

    logout: () => Promise<{ ok: boolean }>;
