    **Step 2: Create `launcher/src/renderer/src/hooks/useSkinHead.ts`:**

    ```typescript
    /**
     * useSkinHead — mc-heads.net avatar URL + placeholder fallback.
     *
     * Per UI-SPEC §Skin avatar contract (D-14): mc-heads.net/avatar/{uuid}/32.
     * Disk cache is deferred to v0.2+ (Phase 2 frozen IPC surface has no file channel).
     * On fetch failure the consumer calls markFetchFailed; the hook then returns
     * the placeholder view permanently for this session.
     */

    import { useEffect, useState } from 'react'

    const failedUuids = new Set<string>()

    export interface SkinHeadView {
      src: string | null
      isPlaceholder: boolean
      initial: string
      markFetchFailed: () => void
    }

    export function useSkinHead(
      uuid: string | undefined,
      username: string | undefined
    ): SkinHeadView {
      const [failed, setFailed] = useState(false)

      // Reset per-uuid when uuid changes
      useEffect(() => {
        if (uuid && failedUuids.has(uuid)) {
          setFailed(true)
        } else {
          setFailed(false)
        }
      }, [uuid])

      const initial = deriveInitial(username)

      if (!uuid || failed) {
        return {
          src: null,
          isPlaceholder: true,
          initial,
          markFetchFailed: () => {
            if (uuid) failedUuids.add(uuid)
            setFailed(true)
          }
        }
      }

      return {
        src: `https://mc-heads.net/avatar/${uuid}/32`,
        isPlaceholder: false,
        initial,
        markFetchFailed: () => {
          failedUuids.add(uuid)
          setFailed(true)
        }
      }
    }

    function deriveInitial(username: string | undefined): string {
      if (!username || username.length === 0) return '?'
      return username[0].toUpperCase()
    }

    // Exposed for tests — resets the module-level failed set.
    export const __test__ = {
      resetFailed: (): void => {
        failedUuids.clear()
      }
    }
    ```

    import { describe, it, expect, beforeEach } from 'vitest'

    beforeEach(() => {

    /**
     * DeviceCodeModal (D-06, D-07, UI-SPEC §DeviceCodeModal).
     *
     * Two visible states:
     *   - Active: waiting for Microsoft (shows code + copy + open-in-browser + countdown)
     *   - Expired: timer hit 0, "Generate new code" path
     *
     * Cancellation (ESC or Stop signing in) calls auth store's cancelLogin which
     * hits the frozen auth:logout channel — Plan 03's handler aborts the pending
     * login AND clears state.
     *
     * Copy: verbatim from UI-SPEC §DeviceCodeModal.
     */

    import { useEffect, useRef, useState } from 'react'
    import { Copy, Check, ExternalLink } from 'lucide-react'
    import {
      Dialog,
      DialogContent,
      DialogDescription,
      DialogHeader,
      DialogTitle
    } from '@/components/ui/dialog'
    import { Button } from '@/components/ui/button'
    import { useAuthStore } from '../stores/auth'

    function formatMMSS(ms: number): string {
      const totalSec = Math.max(0, Math.floor(ms / 1000))
      const mm = String(Math.floor(totalSec / 60)).padStart(2, '0')
      const ss = String(totalSec % 60).padStart(2, '0')
      return `${mm}:${ss}`
    }

    function spaceEvery4(s: string): string {
      return s.replace(/(.{4})/g, '$1 ').trim()
    }

    export function DeviceCodeModal(): React.JSX.Element | null {
      const deviceCode = useAuthStore((s) => s.deviceCode)
      const state = useAuthStore((s) => s.state)
      const login = useAuthStore((s) => s.login)
      const cancelLogin = useAuthStore((s) => s.cancelLogin)

      const [copied, setCopied] = useState(false)
      const [remainingMs, setRemainingMs] = useState<number>(() =>
        deviceCode
          ? deviceCode.expiresInSec * 1000 - (Date.now() - deviceCode.receivedAt)
          : 0
      )
      const codeBlockRef = useRef<HTMLDivElement | null>(null)

      // Countdown tick
      useEffect(() => {
        if (!deviceCode) return
        const id = setInterval(() => {
          setRemainingMs(
            deviceCode.expiresInSec * 1000 -
              (Date.now() - deviceCode.receivedAt)
          )
        }, 500)
        return () => clearInterval(id)
      }, [deviceCode])

      // Focus code block on open (screen reader announcement)
      useEffect(() => {
        if (deviceCode && codeBlockRef.current) {
          codeBlockRef.current.focus()
        }
      }, [deviceCode])

      const open = state === 'logging-in' && !!deviceCode

      const handleCopy = async (): Promise<void> => {
        if (!deviceCode) return
        try {
          await navigator.clipboard.writeText(deviceCode.userCode)
          setCopied(true)
          setTimeout(() => setCopied(false), 1500)
        } catch {
          // Clipboard permission denied — silent fail; user can type the code instead
        }
      }

      const handleOpenBrowser = (): void => {
        if (!deviceCode) return
        window.open(deviceCode.verificationUri, '_blank', 'noopener')
      }

      const handleCancel = (): void => {
        void cancelLogin()
      }

      const expired = remainingMs <= 0

      if (!deviceCode) return null

      return (
        <Dialog
          open={open}
          onOpenChange={(nextOpen) => {
            if (!nextOpen) handleCancel()
          }}
        >
          <DialogContent className="sm:max-w-[480px] bg-neutral-900 border-neutral-800">
            <DialogHeader>
              <DialogTitle className="text-2xl font-semibold">
                Sign in with Microsoft
              </DialogTitle>
              <DialogDescription className="text-sm font-normal text-neutral-400">
                {expired
                  ? 'The code expired before you finished signing in.'
                  : "Enter this code on Microsoft's sign-in page to finish logging in."}
              </DialogDescription>
            </DialogHeader>

            {!expired ? (
              <>
                <div className="flex flex-col items-center gap-2 py-2">
                  <span className="text-xs font-normal text-neutral-500">
                    Your code
                  </span>
                  <div
                    ref={codeBlockRef}
                    tabIndex={-1}
                    aria-live="polite"
                    aria-label={`Your sign-in code: ${spaceEvery4(deviceCode.userCode)}`}
                    className="text-2xl font-semibold font-mono tracking-[0.15em] text-neutral-100 select-all"
                  >
                    {deviceCode.userCode}
                  </div>
                </div>

                <div className="flex gap-2 justify-center py-2">
                  <Button
                    size="sm"
                    onClick={() => void handleCopy()}
                    className="bg-[#16e0ee] hover:bg-[#14c9d6] text-neutral-950 text-sm font-normal"
                  >
                    {copied ? (
                      <>
                        <Check className="size-4 mr-2" aria-hidden="true" />
                        Copied
                      </>
                    ) : (
                      <>
                        <Copy className="size-4 mr-2" aria-hidden="true" />
                        Copy code
                      </>
                    )}
                  </Button>

                  <Button
                    size="sm"
                    variant="outline"
                    onClick={handleOpenBrowser}
                    className="text-sm font-normal"
                  >
                    <ExternalLink className="size-4 mr-2" aria-hidden="true" />
                    Open in browser
                  </Button>
                </div>

                <div className="flex items-center justify-between pt-2">
                  <span className="text-xs font-normal text-neutral-500">
                    Expires in {formatMMSS(remainingMs)}
                  </span>
                  <Button
                    size="sm"
                    variant="outline"
                    onClick={handleCancel}
                    className="text-sm font-normal"
                  >
                    Stop signing in
                  </Button>
                </div>
              </>
            ) : (
              <div className="flex justify-end gap-2 pt-4">
                <Button
                  size="sm"
                  variant="outline"
                  onClick={handleCancel}
                  className="text-sm font-normal"
                >
                  Stop signing in
                </Button>
                <Button
                  size="sm"
                  onClick={() => void login()}
                  className="bg-[#16e0ee] hover:bg-[#14c9d6] text-neutral-950 text-sm font-normal"
                >
                  Generate new code
                </Button>
              </div>
            )}
          </DialogContent>
        </Dialog>
      )
    }
